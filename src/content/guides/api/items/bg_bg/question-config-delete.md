Този маршрут предоставя премахване на `QuestionConfig` по id.

**Това ще изтрие всички съответни резултати от въпроси (но не и коментарите).** Това е част от високата цена на кредитите.

[inline-code-attrs-start title = 'Пример за премахване на QuestionConfig с cURL'; type = 'bash'; useDemoTenant = true; isFunctional = false; inline-code-attrs-end]
[inline-code-start]
curl --request DELETE \
  --url 'https://fastcomments.com/api/v1/question-configs/xyz?tenantId=demo&API_KEY=DEMO_API_SECRET'
[inline-code-end]
