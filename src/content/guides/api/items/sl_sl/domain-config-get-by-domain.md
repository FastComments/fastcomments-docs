Posamezne DomainConfigs je mogoče pridobiti po ustrezni `domain`. 

[inline-code-attrs-start title = 'Primer cURL zahteve za konfiguracijo domene'; type = 'bash'; useDemoTenant = true; isFunctional = false; inline-code-attrs-end]
[inline-code-start]
curl --request GET \
  --url 'https://fastcomments.com/api/v1/domain-configs/example.com?tenantId=demo&API_KEY=DEMO_API_SECRET'
[inline-code-end]
