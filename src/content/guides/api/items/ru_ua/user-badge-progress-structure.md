`UserBadgeProgress` — это объект, который представляет прогресс пользователя в получении различных значков в системе FastComments.

Это отслеживание помогает определять, когда пользователи должны получать автоматические значки на основе их активности и участия в вашем сообществе.

Структура объекта `UserBadgeProgress` выглядит следующим образом:

[inline-code-attrs-start title = 'Структура UserBadgeProgress'; type = 'typescript'; isFunctional = false; inline-code-attrs-end]
[inline-code-start]
export interface UserBadgeProgress {
    /** Уникальный идентификатор этой записи прогресса */
    id: string
    /** ID тенанта, которому принадлежит эта запись прогресса */
    tenantId: string
    /** ID пользователя, за которым отслеживается эта запись прогресса */
    userId: string
    /** ID первого комментария пользователя в системе */
    firstCommentId?: string
    /** Дата первого комментария пользователя (миллисекунды с начала эпохи Unix) */
    firstCommentDate?: number
    /** Автоматически рассчитанный коэффициент доверия на основе активности пользователя */
    autoTrustFactor?: number
    /** Коэффициент доверия, установленный вручную администраторами */
    manualTrustFactor?: number
    /** Подробный объект прогресса с различными метриками, ключи соответствуют перечислению BadgeType */
    progress: {
        /** 0: CommentCount - Количество комментариев, оставленных пользователем */
        '0'?: number
        /** 1: CommentUpVotes - Количество полученных пользователем голосов 'за' */
        '1'?: number
        /** 2: CommentReplies - Количество ответов, оставленных пользователем */
        '2'?: number
        /** 3: CommentsPinned - Количество закреплённых комментариев у пользователя */
        '3'?: number
        /** 4: Veteran - Возраст аккаунта пользователя */
        '4'?: number
        /** 5: NightOwl - Количество раз, когда пользователь публиковал в ночное время */
        '5'?: number
        /** 6: FastReplyTime - Среднее время ответа в миллисекундах */
        '6'?: number
        /** 7: ModeratorCommentsDeleted - Для бейджей модератора, количество удалённых комментариев */
        '7'?: number
        /** 8: ModeratorCommentsApproved - Для бейджей модератора, количество одобренных комментариев */
        '8'?: number
        /** 9: ModeratorCommentsUnapproved - Для бейджей модератора, количество неодобренных комментариев */
        '9'?: number
        /** 10: ModeratorCommentsReviewed - Для бейджей модератора, количество проверенных комментариев */
        '10'?: number
        /** 11: ModeratorCommentsMarkedSpam - Для бейджей модератора, количество помеченных как спам комментариев */
        '11'?: number
        /** 12: ModeratorCommentsMarkedNotSpam - Для бейджей модератора, количество помеченных как 'не спам' комментариев */
        '12'?: number
        /** 13: RepliedToSpecificPage - Для каждой страницы, количество ответов */
        '13'?: Record<string, number>
    }
}
